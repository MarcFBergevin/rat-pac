#Add the fork to my local rat-pac depository (once)
git remote add fork https://github.com/MarcFBergevin/rat-pac.git

#First step in developing new code (always):
git checkout master

#Do once step, remote add from main repository (only once, to set up main):
git remote add main https://github.com/rat-pac/rat-pac.git

#Second step (get latest and greatest and dump in main):
git pull main master

#Create an environment to develop code
git branch MarcDev
git checkout MarcDev

#While developing:
git add data/Watchman/Watchman.geo (or file that is to be committed)

#Commit often, commit, seriously.
git commit

#It’s showtime, push to my fork (Ben and Andy)
git push fork MarcDev

#After this is done, open pull request on github.

####### Things to remember

#If you mess up with the commit. Make changes. Unstage file
git commit —amend (overwrite last commit, in case of goof)
